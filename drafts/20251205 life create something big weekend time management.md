        player.fatigue = (player.fatigue || 0) + 1;
        player.heroHealth -= player.fatigue;
        continue;

  function startTurn(state, playerId) {
    const player = state.players[playerId];
    player.maxMana = Math.min(10, player.maxMana + 1);
    player.currentMana = player.maxMana;
    player.heroPowerUsed = false;

  function endTurn(state) {
    if (state.winner) return;

    state.currentPlayerId = next;
    if (next === "player") state.turn += 1;
    startTurn(state, next);
    // auto-run AI if it's AI turn
    if (next === "ai") {

    }

  function dealDamage(state, target, amount) {
    // target: {type:'hero'|'minion', id, owner?}
    if (target.type === "hero") {

      state.log.push(`${amount} damage dealt to ${target.id} (hero).`);

      if (!m) return;
      m.health -= amount;

  function handleMinionDeath(state, ownerPlayer, minion) {
    // remove

  function playCard(state, playerId, handIndex, target) {
    const player = state.players[playerId];
    const card = player.hand[handIndex];
    if (!card) return;
    if (card.cost > player.currentMana)
      return state.log.push("Not enough mana");

    player.currentMana -= card.cost;
    player.hand.splice(handIndex, 1);

    if (card.type === "minion") {
      const minion = {
        ...card,
        health: card.health,
        attack: card.attack,
        instanceId: `${card.id}_${state.instanceCounter++}`,
        canAttack: false,
      };
      player.board.push(minion);
      state.log.push(`${playerId} played minion ${card.name}.`);
    } else if (card.type === "spell") {
      state.log.push(`${playerId} cast ${card.name}.`);
      // resolve built-in spell names
      if (card.name === "Fireball") {
        if (!target) return state.log.push("No target for Fireball");
        dealDamage(state, target, 6);
      } else if (card.name === "Arcane Intellect") {
        drawCard(player, 2);
      } else if (card.name === "Healing Touch") {
        heal(state, playerId, 8);
      } else if (card.name === "Swipe") {
        // do swipe implementation
        dealDamage(
          state,
          { type: "hero", id: opponentIdOf(state, playerId) },
          3
        );
        state.players[opponentIdOf(state, playerId)].board.forEach((m) =>
          dealDamage(
            state,
            {
              type: "minion",
              id: m.instanceId,
              owner: opponentIdOf(state, playerId),
            },
            1
          )
        );
      }
    }
    checkWinner(state);
  }

        `${minion.name} attacked hero ${defenderTarget.id} for ${minion.attack}.`

    minion.canAttack = false;
    checkWinner(state);
