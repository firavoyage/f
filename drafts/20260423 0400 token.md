<script type="module">

function log(msg) {
  console.log(msg)
}

function get_el(id) {
  return document.getElementById(id)
}

function get_number(value) {
  return Number(value)
}

function compute_cost(params) {
  var text = params.text
  var usd_per_million = params.usd_per_million
  var chars_per_token = params.chars_per_token
  var model_price = params.model_price
  var mode_ratio = params.mode_ratio

  var char_count = text.length
  var tokens = char_count / chars_per_token

  var adjusted_tokens = tokens * mode_ratio

  var price_from_model = model_price
  var price_from_input = usd_per_million

  var final_price_per_million = (price_from_model + price_from_input) / 2

  var cost = (adjusted_tokens / 1000000) * final_price_per_million

  return {
    char_count: char_count,
    tokens: tokens,
    adjusted_tokens: adjusted_tokens,
    cost: cost
  }
}

function render() {
  var text_el = get_el("input_text")
  var usd_el = get_el("usd_per_million_input")
  var model_el = get_el("model_select")
  var chars_el = get_el("chars_per_token_input")
  var mode_el = get_el("mode_select")
  var out_el = get_el("output_result")

  var text = text_el.value

  var usd_per_million = get_number(usd_el.value)
  var model_price = get_number(model_el.value)
  var chars_per_token = get_number(chars_el.value)
  var mode_ratio = get_number(mode_el.value)

  var result = compute_cost({
    text: text,
    usd_per_million: usd_per_million,
    chars_per_token: chars_per_token,
    model_price: model_price,
    mode_ratio: mode_ratio
  })

  var output_text =
    "chars: " + result.char_count + "\n" +
    "tokens: " + result.tokens + "\n" +
    "adjusted_tokens: " + result.adjusted_tokens + "\n" +
    "cost_usd: " + result.cost

  out_el.textContent = output_text

  log("render updated")
}

function bind_events() {
  var ids = [
    "input_text",
    "usd_per_million_input",
    "model_select",
    "chars_per_token_input",
    "mode_select"
  ]

  var i = 0
  while (i < ids.length) {
    var el = get_el(ids[i])
    el.addEventListener("input", function () {
      render()
    })
    el.addEventListener("change", function () {
      render()
    })
    i = i + 1
  }
}

function init() {
  bind_events()
  render()
  log("initialized")
}

init()

</script>

</body>
</html>

# 3

what does adjusted_tokens mean

---

`adjusted_tokens` is the **effective token count after applying the “mode multiplier”** (chat / coding / cjk).

In your code:

1. First you estimate tokens:

* `tokens = char_count / chars_per_token`

2. Then you adjust them:

* `adjusted_tokens = tokens * mode_ratio`

So it represents:

> “How many tokens the workload is treated as under different task types”

### Why it exists

Different workloads change token “density”:

* **chat (4)** → normal conversational text (baseline-ish)
* **coding (3)** → code is denser → fewer chars per meaning → effectively more token-heavy
* **cjk (0.7)** → Chinese/Japanese/Korean text packs more meaning per token → effectively cheaper

### Intuition

* If mode_ratio > 1 → you **inflate token usage** (more expensive workload)
* If mode_ratio < 1 → you **deflate token usage** (cheaper workload)

### Simple example

If:

* tokens = 1000
* mode_ratio = 0.7 (cjk)

Then:

* adjusted_tokens = 700

So the system assumes: *“this text behaves like 700 normal tokens worth of cost instead of 1000”*

# 4

remove adjusted tokens. if there is 1000 characters, and i choose 1 token 4 char, then it's simply 250 tokens.

So now:

* **tokens = text.length / chars_per_token**
* no `adjusted_tokens`
