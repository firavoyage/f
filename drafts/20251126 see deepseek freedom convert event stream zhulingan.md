      const inputArea = document.getElementById("input-area");
      const outputArea = document.getElementById("output-area");
      const copyBtn = document.getElementById("copy-btn");

      // Logic to extract text from the specific SSE JSON structure
      function parseStreamData(rawText) {
        let fullMarkdown = "";
        const lines = rawText.split("\n");

        for (const line of lines) {
          // Only process lines starting with "data: "
          if (!line.trim().startsWith("data: ")) continue;

          try {
            // Remove "data: " prefix and parse JSON
            const jsonStr = line.replace(/^data: /, "").trim();
            if (!jsonStr) continue;

            const data = JSON.parse(jsonStr);

            // 1. Check specifically for TEMPLATE_RESPONSE and skip strict matches
            // We check the raw JSON string for the type flag to catch nested structures easily
            if (JSON.stringify(data).includes("TEMPLATE_RESPONSE")) {
              continue;
            }

            // 2. Extraction Heuristic
            // The input has simple strings in 'v' OR complex updates.
            // Based on the example, the content builds up via simple {"v": "string"} objects
            // and sometimes via path updates.

            // Direct string value (most common in the log provided)
            if (typeof data.v === "string") {
              fullMarkdown += data.v;
            }

            // If 'v' is an array (complex update), we usually ignore it for simple text extraction
            // unless we want to parse the deep diff structure.
            // Given the example log, the text flows through the simple "v": "string" lines.
          } catch (e) {
            // Ignore parse errors (e.g. "data: [DONE]")
          }
        }
        return fullMarkdown;
      }

        const rawInput = inputArea.value;
        const markdownText = parseStreamData(rawInput);

        // Render HTML
        outputArea.innerHTML = marked.parse(markdownText);

          try {

            // Visual Feedback

              copyBtn.innerText = "Copy Markdown";
              copyBtn.style.backgroundColor = "";
            }, 2000);
          } catch (err) {
            console.error("Failed to copy", err);
            copyBtn.innerText = "Copy Failed";
          }
        }
      }

      // Manual Copy Button logic
      copyBtn.addEventListener("click", async () => {
        const markdownText = parseStreamData(inputArea.value);
        await navigator.clipboard.writeText(markdownText);
        copyBtn.innerText = "Copied Manually!";
        setTimeout(() => {
          copyBtn.innerText = "Copy Markdown";
        }, 2000);
      });

      // Listen for paste/input
      inputArea.addEventListener("input", handleSync);
    </script>
